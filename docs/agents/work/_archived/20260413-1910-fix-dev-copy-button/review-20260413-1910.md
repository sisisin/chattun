   ```tsx
   // 修正例
   const handleClick = () => {
     copyClicked(message);
     setCopied(true);
   };

   React.useEffect(() => {
     if (!copied) return;
     const timerId = setTimeout(() => setCopied(false), 1500);
     return () => clearTimeout(timerId);
   }, [copied]);
   ```
